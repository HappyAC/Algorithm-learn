bool bellman_ford(int s) 
{	// n个顶点，m条边
   	fill(dist, dist+n+1, INT_MAX);  dist[s]=0;	 
	int i;
   	for(i=1; i<=n-1; i++) 
	{// i 控制外循环次数，无其他含义
		for(int j=0; j<m; j++)
			if(dist[e[j].x]!=INT_MAX) dist[e[j].y]=Min(dist[e[j].y],dist[e[j].x]+e[j].w);		
	}
	for(i=1;i<=n;i++)
	{
		ans[s][i]=dist[i];
		ans[i][s]=dist[i];
	}
	for(i=0; i<m; i++)
		if(dist[e[i].y]>dist[e[i].x]+e[i].w) return false; // 有负环
		
		return true;	// 无负环
}
