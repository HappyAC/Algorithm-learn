#include<stdio.h>//n个中m个错排的方法数
__int64 c(__int64 n,__int64 m)
{
	if(m==0||n==m)
		return 1;
	else
		return c(n-1,m)+c(n-1,m-1);
}
int main()
{
	__int64 a[22],i,n,m;
	a[1]=0;a[2]=1;
	for(i=3;i<=21;i++)
		a[i]=(i-1)*(a[i-1]+a[i-2]);
	__int64 t;
	scanf("%I64d",&t);
	while(t--)
	{
		scanf("%I64d %I64d",&n,&m);
		printf("%I64d\n",c(n,m)*a[m]);
	}
	return 0;
}
