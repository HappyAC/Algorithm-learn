int extend_gcd(int a,int b,int &x,int &y)
{
    if(a == 0 && b == 0) return -1;
    if(b == 0){x = 1; y = 0; return a;}
    int d = extend_gcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}

int mod_reverse(int a, int n)
{
    int x,y;
    int d = extend_gcd(a, n, x, y);
    if(d == 1) return (x % n + n) % n;
    else return -1;
}
