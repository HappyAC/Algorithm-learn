#include <iostream>
using namespace std;

int n,k;
int dp[222][10];

bool run()
{
    if(!(cin>>n>>k)) return false;
    dp[0][0] = 1;
    for(int i=1; i<=n; i++)
    {
        for(int j=1; j<=k; j++)
        {
            if(i>=j) dp[i][j]=dp[i-j][j]+dp[i-1][j-1];
        }
    }
    cout<<dp[n][k]<<endl;
	return true;
}
int main()
{
	while(run()) ;
	return 0;
}
