void EK()
{
	queue<int>q;
	fill(&flow[0][0],&flow[N-1][N-1]+1,0);
	f=0;
	for(;;)
	{
		fill(a,a+N,0);
		a[s]=INF;
		q.push(s);
		while(!q.empty())
		{
			int u=q.front();
			q.pop();
			for(int v=1;v<=n;v++)
			{
				if(!a[v] && cap[u][v]>flow[u][v])
				{
					p[v]=u;
					q.push(v);
					a[v]=MIN(a[u],cap[u][v]-flow[u][v]);
				}
			}
		}
		if(a[t]==0) break;
		for(int u=t;u!=s;u=p[u])
		{
			flow[p[u]][u] += a[t];
			flow[u][p[u]] -= a[t];
		}
		f += a[t];
	}
}
